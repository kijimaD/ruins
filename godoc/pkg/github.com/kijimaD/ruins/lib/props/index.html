type PropConfig ¶

PropConfig は置物の設定情報を保持する 

type PropConfig struct {
    Name             string      // 置物の名前
    Type             gc.PropType // 置物のタイプ
    SpriteNumber     int         // スプライト番号
    BlocksMovement   bool        // 移動を阻害するか
    BlocksVisibility bool        // 視界を遮るか
    SpawnWeight      int         // ランダム配置での重み
    Description      string      // 説明文
}
